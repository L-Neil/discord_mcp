    --from-literal=DISCORD_BOT_TOKEN='<bot-token>' \
    --from-literal=DISCORD_GUILD_ID='<guild-id>' \
    --from-literal=MCP_AUTH_TOKENS='<tok1,tok2,...>'

  kubectl -n {{ .Release.Namespace }} rollout status deploy/{{ include "discord-mcp.fullname" . }}
